        // --- 系统设置与 API 拉取 ---
        async function pullModels() {
            const base = document.getElementById('api-base').value.trim();
            const key = document.getElementById('api-key').value.trim();
            const btn = document.getElementById('btn-pull-models');
            
            if (!base || !key) {
                alert("请先填写 API 地址和 API Key");
                return;
            }
            
            let fetchUrl = base.replace(/\/+$/, '');
            if (!fetchUrl.endsWith('/models')) {
                fetchUrl += (fetchUrl.endsWith('/v1') ? '/models' : '/v1/models');
            }
            
            btn.innerText = "正在拉取...";
            try {
                const response = await fetch(fetchUrl, {
                    method: 'GET',
                    headers: { 'Authorization': `Bearer ${key}` }
                });
                
                if (!response.ok) {
                    const errText = await response.text();
                    throw new Error(`HTTP 错误: ${response.status} - ${errText.substring(0, 50)}`);
                }
                
                const data = await response.json();
                const modelSelect = document.getElementById('api-model');
                
                modelSelect.innerHTML = '';
                
                if (data.data && Array.isArray(data.data)) {
                    data.data.forEach(m => {
                        const opt = document.createElement('option');
                        opt.value = m.id;
                        opt.innerText = m.id;
                        modelSelect.appendChild(opt);
                    });
                    alert(`成功拉取 ${data.data.length} 个模型!`);
                    saveSysSettings();
                } else {
                    alert("获取的数据格式不符合 OpenAI 标准:\n" + JSON.stringify(data).substring(0, 100));
                }
            } catch (err) {
                alert("拉取模型失败 (请检查地址、网络或跨域设置): \n" + err.message);
            } finally {
                btn.innerText = "拉取模型";
            }
        }

        // 核心更新：长按气泡精准定位计算
        function openMsgActionMenu(idx) {
            targetMsgIdx = idx;
            const menu = document.getElementById('msg-action-menu');
            const popup = document.getElementById('action-popup-box');
            
            menu.style.display = 'block'; 
            
            const bubble = document.getElementById('bubble-' + idx);
            if (bubble) {
                const rect = bubble.getBoundingClientRect();
                
                // 预设位置：默认在气泡正下方
                let top = rect.bottom + 8;
                let left = rect.left;
                
                // 获取气泡所在的 wrap 来判断是左边还是右边的消息
                const wrap = bubble.closest('.msg-wrap');
                const isRight = wrap && wrap.classList.contains('msg-right');
                
                // 临时显示以获取 popup 实际宽高
                popup.style.visibility = 'hidden';
                popup.style.left = '0px';
                popup.style.top = '0px';
                
                setTimeout(() => {
                    const popupRect = popup.getBoundingClientRect();
                    
                    if (isRight) {
                        // 右侧消息：菜单右对齐气泡
                        left = rect.right - popupRect.width;
                    }
                    
                    // 屏幕水平边界碰撞检测
                    if (left < 15) left = 15;
                    if (left + popupRect.width > window.innerWidth - 15) {
                        left = window.innerWidth - popupRect.width - 15;
                    }
                    
                    // 垂直碰撞检测：如果下方超出屏幕，显示在气泡上方
                    if (top + popupRect.height > window.innerHeight - 20) {
                        top = rect.top - popupRect.height - 8;
                    }
                    
                    popup.style.left = left + 'px';
                    popup.style.top = top + 'px';
                    popup.style.visibility = 'visible';
                }, 0);
            }
        }

        async function fetchAIResponse(friendIndex) {
            const f = customFriends[friendIndex];
            const configStr = localStorage.getItem('sys_api_config');
            const headerNameEl = document.getElementById('chat-header-name');
            const originalName = f.note || f.name;

            if (activeFriendIndex === friendIndex) {
                headerNameEl.innerText = "对方正在输入中...";
            }

            if (!configStr) {
                setTimeout(() => {
                    f.messages.push({ role: 'ai', type: 'text', text: "系统提示：请先在主页设置中配置有效的 API 地址和 Key。", time: getCurrentTimeStr() });
                    saveFriends();
                    if (activeFriendIndex === friendIndex) {
                        headerNameEl.innerText = originalName;
                        renderMessages();
                    }
                }, 1000);
                return;
            }

            const config = JSON.parse(configStr);
            if (!config.key || !config.base) {
                setTimeout(() => {
                    f.messages.push({ role: 'ai', type: 'text', text: "系统提示：API Key 或地址未配置完整。", time: getCurrentTimeStr() });
                    saveFriends();
                    if (activeFriendIndex === friendIndex) {
                        headerNameEl.innerText = originalName;
                        renderMessages();
                    }
                }, 1000);
                return;
            }

            let messages = [];
            const now = new Date();
            const dateOptions = { year: 'numeric', month: 'long', day: 'numeric', weekday: 'long' };
            const timeOptions = { hour: '2-digit', minute: '2-digit' };
            const currentDateStr = now.toLocaleDateString('zh-CN', dateOptions);
            const currentTimeStr = now.toLocaleTimeString('zh-CN', timeOptions);

            let systemPrompt = `[系统参考] 当前时间：${currentDateStr} ${currentTimeStr} (仅作背景感知，请勿刻意强调)。\n[回复要求] 一般回复3到5句话(除非角色设定中另有明确字数限制，一切以设定的要求为准)。请直接使用角色应有的语言回复，绝对不要在回复中附带中文翻译。\n`;
            if (f.prompt) systemPrompt += `你的角色设定：\n${f.prompt}\n`;
            if (f.persona) systemPrompt += `我的人设（与你对话的人）：\n${f.persona}\n`;
            if (f.worldbook) systemPrompt += `世界观背景（世界书）：\n${f.worldbook}\n`;

            if (systemPrompt.trim()) {
                messages.push({ role: "system", content: systemPrompt });
            }

            const memoryLimit = parseInt(config.memory) || 50;
            const history = f.messages.slice(-memoryLimit);
            history.forEach(m => {
                let apiRole = m.role === 'ai' ? 'assistant' : 'user';
                
                if (m.type === 'image' && m.imgUrl) {
                    messages.push({
                        role: apiRole,
                        content: [
                            { type: "text", text: "请识别并结合人设回复这张图片：" },
                            { type: "image_url", image_url: { url: m.imgUrl } }
                        ]
                    });
                } else if (m.type === 'photo_mock') {
                    messages.push({ role: apiRole, content: `[用户发送了一张照片，内容描述为：${m.text}]` });
                } else if (m.type === 'voice') {
                    messages.push({ role: apiRole, content: `[用户发送一条语音，内容为：${m.text}]` });
                } else {
                    messages.push({ role: apiRole, content: m.text });
                }
            });

            try {
                let fetchUrl = config.base.replace(/\/+$/, '');
                if (!fetchUrl.endsWith('/chat/completions')) {
                    fetchUrl += (fetchUrl.endsWith('/v1') ? '/chat/completions' : '/v1/chat/completions');
                }

                const response = await fetch(fetchUrl, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'Authorization': `Bearer ${config.key}`
                    },
                    body: JSON.stringify({
                        model: config.model || 'gpt-3.5-turbo',
                        messages: messages,
                        temperature: parseFloat(config.temp) || 1.0,
                        top_p: 1.0,
                        frequency_penalty: 0,
                        presence_penalty: 0
                    })
                });

                if (response.ok) {
                    const data = await response.json();
                    if (data.choices && data.choices.length > 0) {
                        const aiText = data.choices[0].message.content.trim();
                        f.messages.push({ role: 'ai', type: 'text', text: aiText, time: getCurrentTimeStr() });
                    } else {
                        f.messages.push({ role: 'ai', type: 'text', text: `[API返回异常: 未找到回复内容]`, time: getCurrentTimeStr() });
                    }
                } else {
                    const errData = await response.text();
                    let errMsg = `[API错误: ${response.status}]`;
                    try {
                        const parsedErr = JSON.parse(errData);
                        if (parsedErr.error && parsedErr.error.message) {
                            errMsg += ` ${parsedErr.error.message}`;
                        }
                    } catch(e) {
                        errMsg += ` ${errData.substring(0, 60)}...`;
                    }
                    f.messages.push({ role: 'ai', type: 'text', text: errMsg, time: getCurrentTimeStr() });
                }
            } catch (err) {
                f.messages.push({ role: 'ai', type: 'text', text: `[网络请求失败: ${err.message}] 请检查API地址或跨域设置。`, time: getCurrentTimeStr() });
            }

            saveFriends();
            if (activeFriendIndex === friendIndex) {
                headerNameEl.innerText = originalName;
                renderMessages();
            }
        }

        async function translateVoiceMsg(idx) {
            const f = customFriends[activeFriendIndex];
            const msg = f.messages[idx];
            if (msg.translatedText) return;

            try {
                const configStr = localStorage.getItem('sys_api_config');
                let translated = "翻译失败：请配置API";
                if (configStr) {
                    const config = JSON.parse(configStr);
                    if (config.key && config.base) {
                        let fetchUrl = config.base.replace(/\/+$/, '');
                        if (!fetchUrl.endsWith('/chat/completions')) {
                            fetchUrl += (fetchUrl.endsWith('/v1') ? '/chat/completions' : '/v1/chat/completions');
                        }
                        const response = await fetch(fetchUrl, {
                            method: 'POST',
                            headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${config.key}` },
                            body: JSON.stringify({
                                model: config.model || 'gpt-3.5-turbo',
                                messages: [{role: 'user', content: `请将以下内容翻译成精准的中文，只需要返回翻译结果，不要任何多余的解释：\n${msg.text}`}],
                                temperature: 0.3
                            })
                        });
                        if (response.ok) {
                            const data = await response.json();
                            translated = data.choices[0].message.content.trim();
                        } else {
                            translated = "翻译失败：API请求错误";
                        }
                    }
                }
                msg.translatedText = translated;
                msg.showTranslation = true;
                saveFriends();
                const transDiv = document.getElementById(`voice-trans-${idx}`);
                if(transDiv) transDiv.innerText = translated;
                else renderMessages();
            } catch (e) {
                msg.translatedText = "翻译出错";
                msg.showTranslation = true;
                saveFriends();
                renderMessages();
            }
        }

        async function handleTranslate(element, msgIdx) {
            const f = customFriends[activeFriendIndex];
            if (!f.autoTranslate) return;
            const msg = f.messages[msgIdx];

            if (msg.translatedText) {
                msg.showTranslation = !msg.showTranslation;
                saveFriends();
                renderMessages();
                return;
            }

            const originalText = msg.text;
            element.innerHTML = `<div class="original-text" style="line-height: 1.5;">${originalText}</div><div class="translate-divider"></div><div class="translated-text">翻译中...</div>`;

            try {
                const configStr = localStorage.getItem('sys_api_config');
                let translated = "";
                if (configStr) {
                    const config = JSON.parse(configStr);
                    if (config.key && config.base) {
                        let fetchUrl = config.base.replace(/\/+$/, '');
                        if (!fetchUrl.endsWith('/chat/completions')) {
                            fetchUrl += (fetchUrl.endsWith('/v1') ? '/chat/completions' : '/v1/chat/completions');
                        }
                        const response = await fetch(fetchUrl, {
                            method: 'POST',
                            headers: {
                                'Content-Type': 'application/json',
                                'Authorization': `Bearer ${config.key}`
                            },
                            body: JSON.stringify({
                                model: config.model || 'gpt-3.5-turbo',
                                messages: [{role: 'user', content: `请将以下内容翻译成精准的中文，只需要返回翻译结果，不要任何多余的解释：\n${originalText}`}],
                                temperature: 0.3
                            })
                        });
                        if (response.ok) {
                            const data = await response.json();
                            translated = data.choices[0].message.content.trim();
                        } else {
                            translated = "翻译失败：API请求错误 (" + response.status + ")";
                        }
                    } else {
                        translated = "翻译失败：请先配置有效的 API Key";
                    }
                } else {
                    translated = "翻译失败：请先配置 API";
                }
                msg.translatedText = translated;
                msg.showTranslation = true; 
                saveFriends();
                renderMessages(); 
            } catch (error) {
                msg.translatedText = "翻译请求出错：" + error.message;
                msg.showTranslation = true;
                saveFriends();
                renderMessages();
            }
        }

        function renderMessages() {

                    const isUser = m.role === 'user';

                        if(f.hideMode === 'both') showAva = false;
                        if(f.hideMode === 'mine' && isUser) showAva = false;

                            if (f.messages[j].role === 'ai') { isMessageRead = true; break; }
                        }
                    }

                    // --- 多选状态相关元素 ---

                    let wrapperClick = '';

                        bubbleContent = `<div onclick="toggleVoice(${idx})" style="display:flex; align-items:center; gap:8px; min-width:80px;">

                            </div>
                        </div>

            }
            container.scrollTop = container.scrollHeight;
        }

        // --- 状态栏功能 ---
        function initStatusBar() {
            const updateTime = () => {
                const now = new Date();
                const hours = String(now.getHours()).padStart(2, '0');
                const minutes = String(now.getMinutes()).padStart(2, '0');
                document.getElementById('status-time').innerText = `${hours}:${minutes}`;
            };
            setInterval(updateTime, 1000);
            updateTime();

            if ('getBattery' in navigator) {
                navigator.getBattery().then(battery => {
                    const updateBattery = () => {
                        const levelEl = document.getElementById('battery-level');
                        const iconEl = document.getElementById('battery-icon');
                        levelEl.style.width = (battery.level * 100) + '%';
                        if (battery.charging) {
                            iconEl.classList.add('is-charging');
                            levelEl.style.background = ''; 
                        } else {
                            iconEl.classList.remove('is-charging');
                            if (battery.level <= 0.2) {
                                levelEl.style.background = '#FF3B30'; 
                            } else {
                                levelEl.style.background = ''; 
                            }
                        }
                    };
                    battery.addEventListener('levelchange', updateBattery);
                    battery.addEventListener('chargingchange', updateBattery);
                    updateBattery();
                });
            } else {
                document.getElementById('battery-level').style.width = '85%';
            }

            const updateNetwork = () => {
                const conn = navigator.connection || navigator.mozConnection || navigator.webkitConnection;
                const typeEl = document.getElementById('status-network');
                if (conn && conn.effectiveType) {
                    let type = conn.effectiveType;
                    if (type === '4g' || type === '5g') {
                        typeEl.innerText = '5G'; 
                    } else if (type === 'wifi') {
                        typeEl.innerText = 'Wi-Fi';
                    } else {
                        typeEl.innerText = type.toUpperCase();
                    }
                } else {
                    typeEl.innerText = '5G';
                }
            };
            if (navigator.connection) {
                navigator.connection.addEventListener('change', updateNetwork);
            }
            updateNetwork();
        }

            ['icon-settings','icon-theme','icon-kakao','icon-ins','icon-bubble','icon-weibo','icon-x','icon-weverse'].forEach(id => {
                const iconData = localStorage.getItem('img_' + id);
                if(iconData) {
                    const homeIcon = document.getElementById(id);
                    if (homeIcon) { homeIcon.style.backgroundImage = `url(${iconData})`; homeIcon.innerHTML = ''; }
                    const prevIcon = document.getElementById('prev-' + id);
                    if (prevIcon) { prevIcon.style.backgroundImage = `url(${iconData})`; prevIcon.innerHTML = ''; }
                }
            });

            // 载入聊天壁纸
            const chatWall = localStorage.getItem('img_chat-wallpaper');
            if (chatWall) {
                document.documentElement.style.setProperty('--chat-bg-img', `url(${chatWall})`);
            }

            const savedKName = localStorage.getItem('kkt_my_name');
            if (savedKName) {
                document.getElementById('k-my-name').innerText = savedKName;
                document.getElementById('txt-nickname').innerText = savedKName;
            } else {
                document.getElementById('k-my-name').innerText = document.getElementById('txt-nickname').innerText;
            }

            document.addEventListener('click', function(event) {
                const menu = document.getElementById('kakao-add-menu');
                const btn = document.querySelector('.kakao-header-v2 .icons svg:nth-child(2)');
                if (menu && menu.style.display === 'block' && !menu.contains(event.target) && event.target !== btn && !btn.contains(event.target)) {
                    menu.style.display = 'none';
                }
            });
        }
    </script>
